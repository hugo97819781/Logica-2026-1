    \begin{ejercicio}{2}
        Al decir la frase \textit{"Consideremos al primer número que no puede ser descrito con menos de quince palabras"} se están utilizando menos de 15 palabras para describir a tal número... Parece haber una paradoja, ¿de dónde surge esta paradoja?, y, ¿cómo se podría solventar?
    \end{ejercicio}

    \begin{ejercicio}{2}
        Sean $X$ un conjunto, $n$ un natural y $R$ una relación de aridad $n$ en $X$. Definase la relación $S$ en $R$ como: $f \mathrel{S} g$ si y sólo si existe $h:n \to n$ biyectiva tal que $f=g \circ h$. Pruebe que $S$ es relación de equivalencia.
    \end{ejercicio}

    \begin{ejercicio}{3}
        Considere el tipo de semejanza $\tau=\{R\}$, donde $R$ es una letra relacional de aridad $2$. Encuentre alguna fórmula el lenguaje de primer orden correspondiente a $\tau$, que sea simultáneamente:
        \begin{enumerate}
            \item Verdadera en la estructura $(\mathbb{N},<)$.
            \item Falsa en la estructura $(\mathscr{P}(\mathbb{N}),\subsetneq)$.
            \item Verdadera en la estructura $(\mathbb{Z},>)$.
            \item Falsa en la estructura $(5 \cup \{7\}, \in)$.
        \end{enumerate}
        Justifique sus afirmaciones. 
    \end{ejercicio}

    \begin{ejercicio}{3}
        Sea $\tau$ el tipo de semejanza $\{c\} \cup \{R\}$, donde, $c$ es una letra de constante y $R$ una letra relacional de aridad $3$. Considere el enunciado:
        \[ \forall x \, \forall y \, \Big( \forall z \big( R(z,x,c) \leftrightarrow R(z,y,c) \big) \to x=y \Big) \]

        De dos ejemplos de estructuras, en la primera, el enunciado dado debe ser verdadero; y, en la segunda, debe ser falso.
    \end{ejercicio}
